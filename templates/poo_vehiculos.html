Programación Orientada a Objetos: Vehículos

Este ejemplo representa una jerarquía de vehículos usando clases, herencia y especialización. Es una forma de estructurar el código para representar distintos tipos de transporte.

Clase base: Vehiculo


class Vehiculo:
    def __init__(self, marca, modelo):
        self.marca = marca
        self.modelo = modelo

    def mostrar_info(self):
        print(f"{self.marca} - {self.modelo}")

Subclase: Embarcacion


from Vehiculos import Vehiculo

class Embarcacion(Vehiculo):
    def __init__(self, marca, modelo, eslora):
        super().__init__(marca, modelo)
        self.eslora = eslora

    def mostrar_info(self):
        super().mostrar_info()
        print(f"Eslora: {self.eslora} metros")

Subclase: Ferri


from Embarcacion import Embarcacion

class Ferri(Embarcacion):
    def __init__(self, marca, modelo, eslora, capacidad_pasajeros):
        super().__init__(marca, modelo, eslora)
        self.capacidad_pasajeros = capacidad_pasajeros

    def mostrar_info(self):
        super().mostrar_info()
        print(f"Capacidad: {self.capacidad_pasajeros} pasajeros")

Uso práctico en main.py


from Ferri import Ferri

f1 = Ferri("Balearia", "FastFerry", 120, 500)
f1.mostrar_info()

Resultado:


Balearia - FastFerry
Eslora: 120 metros
Capacidad: 500 pasajeros

Este ejemplo muestra cómo extender clases para añadir funcionalidades específicas, aplicando el concepto de herencia múltiple y reutilización de código.
